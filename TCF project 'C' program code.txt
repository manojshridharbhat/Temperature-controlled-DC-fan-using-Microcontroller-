void fanon()
{

}
void fanoff()
{

}
unsigned char adc()
{

rd_adc=1; wr_adc=1; 
while(intr_adc==1); rd_adc=0; conv=P1;

}
void datawrt(unsigned char y)
{

}
void lcddata(unsigned char *str) //LCD data Function
{
while(*str != '\0')
{
ldata = *str; rs=1;
rw=0; en=1; 
msdelay(1); 
en=0; 
str++;
}
}
 

void lcdcmd(unsigned char cmd)  //LCD cammand Function
{
ldata=cmd; rs=0; rw=0; en=1;
msdelay(1); en=0;
}
